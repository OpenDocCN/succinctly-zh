                 // OSM Mapnik
                 var osmLink = "<a href='http://www.openstreetmap.org'>Open StreetMap</a>";
                 L.tileLayer(
                     'http://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {

                 var runLayer = omnivore.kml('kml/2.5_month_age.kml')

                             layer.bindPopup(layer.feature.properties.name);
                         });
                     }).addTo(map);
             }
         </script>

             <div id="map"></div>
         </body>
